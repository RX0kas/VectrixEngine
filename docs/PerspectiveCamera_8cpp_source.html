28 void PerspectiveCamera::recalculateViewMatrix()
29 {
30 const float c1 = glm::cos(m_Rotation.y); // yaw
31 const float s1 = glm::sin(m_Rotation.y);
32 const float c2 = glm::cos(m_Rotation.x); // pitch
33 const float s2 = glm::sin(m_Rotation.x);
34 const float c3 = glm::cos(m_Rotation.z); // roll
35 const float s3 = glm::sin(m_Rotation.z);
36
37 const glm::vec3 u{
38 c1 * c3 + s1 * s2 * s3,
39 c2 * s3,
40 c1 * s2 * s3 - c3 * s1
41 };
42
43 const glm::vec3 v{
44 c3 * s1 * s2 - c1 * s3,
45 c2 * c3,
46 c1 * c3 * s2 + s1 * s3
47 };
48
49 const glm::vec3 w{
50 c2 * s1,
51 -s2,
52 c1 * c2
53 };
54
55 m_ViewMatrix = glm::mat4{1.0f};
56 m_ViewMatrix[0][0] = u.x;
57 m_ViewMatrix[1][0] = u.y;
58 m_ViewMatrix[2][0] = u.z;
59 m_ViewMatrix[0][1] = v.x;
60 m_ViewMatrix[1][1] = v.y;
61 m_ViewMatrix[2][1] = v.z;
62 m_ViewMatrix[0][2] = w.x;
63 m_ViewMatrix[1][2] = w.y;
64 m_ViewMatrix[2][2] = w.z;
65 m_ViewMatrix[3][0] = -glm::dot(u, m_Position);
66 m_ViewMatrix[3][1] = -glm::dot(v, m_Position);
67 m_ViewMatrix[3][2] = -glm::dot(w, m_Position);
68 }

75 void PerspectiveCamera::recalculateProjectionMatrix() {
76 float aspect = Application::instance().window().getAspect();
77 VC_CORE_ASSERT(aspect > std::numeric_limits<float>::epsilon(),"Aspect ratio is zero or invalid");
78
79 const float tanHalfFovy = std::tan(FOVY / 2.f);
80 m_ProjectionMatrix = glm::mat4{0.0f};
81 m_ProjectionMatrix[0][0] = 1.f / (aspect * tanHalfFovy);
82 m_ProjectionMatrix[1][1] = 1.f / (tanHalfFovy);
83 m_ProjectionMatrix[2][2] = CAM_FAR / (CAM_FAR - CAM_NEAR);
84 m_ProjectionMatrix[2][3] = 1.f;
85 m_ProjectionMatrix[3][2] = -(CAM_FAR * CAM_NEAR) / (CAM_FAR - CAM_NEAR);
86
87 m_ProjectionMatrix[1][1] *= -1;
88 }
